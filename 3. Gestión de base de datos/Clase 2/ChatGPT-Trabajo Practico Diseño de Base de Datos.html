d) Diagrama de Tablas + Implementación en SQL

Tablas SQL (esquema básico)

sql
CREATE TABLE ModeloTelevisor (
    ID_Modelo INT PRIMARY KEY,
    NombreModelo VARCHAR(100),
    Descripcion TEXT
);

CREATE TABLE Televisor (
    ID_Televisor INT PRIMARY KEY,
    NumeroSerie VARCHAR(100),
    FechaArmado DATE,
    ID_Modelo INT,
    FOREIGN KEY (ID_Modelo) REFERENCES ModeloTelevisor(ID_Modelo)
);

CREATE TABLE Pieza (
    ID_Pieza INT PRIMARY KEY,
    NombrePieza VARCHAR(100),
    Tipo ENUM('Importada', 'Fabricada'),
    Costo DECIMAL(10,2)
);

CREATE TABLE Operario (
    ID_Operario INT PRIMARY KEY,
    Nombre VARCHAR(100),
    Apellido VARCHAR(100),
    DNI VARCHAR(20)
);

CREATE TABLE HojaConfeccion (
    ID_Hoja INT PRIMARY KEY,
    ID_Pieza INT,
    ID_Operario INT,
    Fecha DATE,
    CantidadFabricada INT,
    FOREIGN KEY (ID_Pieza) REFERENCES Pieza(ID_Pieza),
    FOREIGN KEY (ID_Operario) REFERENCES Operario(ID_Operario)
);

CREATE TABLE Importador (
    ID_Importador INT PRIMARY KEY,
    Nombre VARCHAR(100),
    País VARCHAR(50)
);

CREATE TABLE FacturaCompra (
    ID_Factura INT PRIMARY KEY,
    Fecha DATE,
    ID_Importador INT,
    ID_Pieza INT,
    Cantidad INT,
    PrecioUnitario DECIMAL(10,2),
    FOREIGN KEY (ID_Importador) REFERENCES Importador(ID_Importador),
    FOREIGN KEY (ID_Pieza) REFERENCES Pieza(ID_Pieza)
);

CREATE TABLE MapaArmado (
    ID_Modelo INT,
    ID_Pieza INT,
    OrdenEnArmado INT,
    Ubicacion VARCHAR(100),
    PRIMARY KEY (ID_Modelo, ID_Pieza),
    FOREIGN KEY (ID_Modelo) REFERENCES ModeloTelevisor(ID_Modelo),
    FOREIGN KEY (ID_Pieza) REFERENCES Pieza(ID_Pieza)
);

e) Consultas para probar la Base de Datos

1. Obtener todas las piezas y su ubicación en el modelo de televisor "SMART-LED-42"

sql
SELECT p.NombrePieza, m.OrdenEnArmado, m.Ubicacion
FROM MapaArmado m
JOIN Pieza p ON m.ID_Pieza = p.ID_Pieza
JOIN ModeloTelevisor mt ON m.ID_Modelo = mt.ID_Modelo
WHERE mt.NombreModelo = 'SMART-LED-42'
ORDER BY m.OrdenEnArmado;

2. Listar los operarios que fabricaron más de 100 piezas en total

sql
SELECT o.Nombre, o.Apellido, SUM(h.CantidadFabricada) AS TotalFabricado
FROM HojaConfeccion h
JOIN Operario o ON h.ID_Operario = o.ID_Operario
GROUP BY o.ID_Operario
HAVING SUM(h.CantidadFabricada) > 100;

Mientras tanto, te dejo el código que podés ejecutar vos mismo si tenés instalado Python con la biblioteca graphviz. Este código genera el Diagrama Entidad-Relación (DER) en formato imagen:

🐍 Código en Python para generar el DER:

python
import graphviz

# Crear un Diagrama Entidad-Relación
der = graphviz.Digraph(format='png')
der.attr(rankdir='LR', size='8,5')

# Entidades y atributos
entities = {
    "Televisor": ["ID_Televisor (PK)", "NúmeroSerie", "FechaArmado", "ID_Modelo (FK)"],
    "ModeloTelevisor": ["ID_Modelo (PK)", "NombreModelo", "Descripción"],
    "Pieza": ["ID_Pieza (PK)", "NombrePieza", "Tipo", "Costo"],
    "Operario": ["ID_Operario (PK)", "Nombre", "Apellido", "DNI"],
    "HojaConfeccion": ["ID_Hoja (PK)", "ID_Pieza (FK)", "ID_Operario (FK)", "Fecha", "CantidadFabricada"],
    "Importador": ["ID_Importador (PK)", "Nombre", "País"],
    "FacturaCompra": ["ID_Factura (PK)", "Fecha", "ID_Importador (FK)", "ID_Pieza (FK)", "Cantidad", "PrecioUnitario"],
    "MapaArmado": ["ID_Modelo (FK)", "ID_Pieza (FK)", "OrdenEnArmado", "Ubicación"]
}

# Agregar entidades
for entity, attributes in entities.items():
    label = f"{entity}\n" + "\n".join(attributes)
    der.node(entity, label=label, shape="record")

# Relaciones (entidad origen, entidad destino, cardinalidad)
relationships = [
    ("ModeloTelevisor", "Televisor", "1", "N"),
    ("ModeloTelevisor", "MapaArmado", "1", "N"),
    ("Pieza", "MapaArmado", "1", "N"),
    ("Pieza", "HojaConfeccion", "1", "N"),
    ("Operario", "HojaConfeccion", "1", "N"),
    ("Importador", "FacturaCompra", "1", "N"),
    ("Pieza", "FacturaCompra", "1", "N")
]

# Agregar relaciones
for src, dst, src_mult, dst_mult in relationships:
    der.edge(src, dst, label=f"{src_mult}:{dst_mult}")

# Guardar y renderizar
der.render("DER_Televisores", view=True)
